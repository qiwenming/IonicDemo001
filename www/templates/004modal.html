      /路由设置
      .config(function($stateProvider,$urlRouterProvider){
        $stateProvider
          .state('004',{
            url:'/004',
            templateUrl:"templates/004modal.html",
            controller:"AppCtrl"
          });
        $urlRouterProvider.otherwise('004');
      });
